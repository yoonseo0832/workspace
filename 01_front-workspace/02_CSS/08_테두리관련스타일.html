테두리 관련 스타일 (border~)

        선택자 {
            /* 테두리 스타일(종류) */
            border[-위치]-style: none | solid | dotted | dashed | doubled | groove | ridge | inset | outset;

            /* 테두리 두께 */
            border[-위치]-width: 값;

            /* 테두리 색상 */
            bordr[-위치]-color: 색상 정보;

            /* 위의 항목들(스타일/두께/색상) 한 번에 적용 */
            border[-위치]: 두께 스타일 [색상];

            /* 테두리 모서리 부분을 둥글게 표현 */
            border[-상하위치 -좌우위치]-radius: 값;
            ex) border-top-left/border-bottom-right
            
            /* 테두리(박스) 그림자 효과 */
            box-shadow: 가로거리(x) 세로거리(y) 흐림정도 번짐정도 색상;
        }
